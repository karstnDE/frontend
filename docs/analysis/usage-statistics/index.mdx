import BrowserOnly from '@docusaurus/BrowserOnly';
import LoadingSpinner from '@site/src/components/common/LoadingSpinner';

# Wallet Usage

<BrowserOnly fallback={<LoadingSpinner />}>
  {() => {
    const UsageTimeSeriesChart = require('@site/src/components/Usage/UsageTimeSeriesChart').default;
    const UsageTopAddresses = require('@site/src/components/Usage/UsageTopAddresses').default;
    const { useUsageMetrics } = require('@site/src/hooks/useUsageMetrics');

    function UsageStatisticsPage() {
      const { data, loading, error } = useUsageMetrics();

      if (loading) {
        return <LoadingSpinner />;
      }

      if (error || !data) {
        return (
          <div className="alert alert--danger" role="alert">
            <strong>Could not load usage metrics.</strong> {error || 'Unknown error'}
          </div>
        );
      }

      const { summary, daily_users, weekly_users } = data;

      return (
        <>
          <p>
            Wallet usage metrics track how the DefiTuna community interacts with the protocol. Daily active wallets capture
            distinct addresses executing fee-generating actions each day, while weekly active wallets measure the rolling
            seven-day union to highlight sustained participation.
          </p>

          <UsageTimeSeriesChart
            title="Daily Active Wallets"
            data={daily_users.daily_counts}
            yAxisLabel="Daily active wallets"
            description="Number of distinct wallets that executed fee-generating actions each day."
          />

          <UsageTimeSeriesChart
            title="Weekly Active Wallets"
            data={weekly_users.rolling_counts}
            yAxisLabel="Active wallets (7-day)"
            description="Rolling 7-day union of active addresses. This metric identifies sustained protocol use beyond single-day spikes."
          />

          <UsageTopAddresses
            title="Top Daily Users (overall period)"
            rows={daily_users.top_wallets}
            description="Addresses ranked by the number of distinct days they executed fee-generating transactions."
          />
        </>
      );
    }

    return <UsageStatisticsPage />;
  }}
</BrowserOnly>
